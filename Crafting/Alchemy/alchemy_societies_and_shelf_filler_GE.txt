#example: make greater explosion, fill a keg, drop the keg on your shelf.
###Instructions###
#1.)configure shelf profile to pull sulfurous ash, bottles and mortars
#2.)place empty keg in backpack
#3.)on the Scripts>Options tab create a variable named 'empty_keg' and target the keg
#4.)on the Scripts>Options tab create a variable named 'shelf' and target your shelf
#5.)stand in front of shelf and make cure potions
###Instructions###
#make and shelve greater cure 

#check for mortars
if not findtype 'mortar and pestle'
    overhead 'Out of tools!'
    stop
endif
#check for Sulfurous Ash
if count 'Sulfurous Ash' < 6
    overhead 'Out of reagent!'
    menu shelf 1
endif
#stop when society job completed
if insysmsg 'have completed'
    overhead 'Society job completed!' 88
    lifttype 'keg' 1
    drop 'shelf' -1 -1 0
    overhead 'keg dropped'
    stop
endif
#drop a pot on the keg to get started
if findtype 'Purple Potion' true
    lifttype 'Purple Potion' 1
    drop 'empty_keg' -1 -1 0
    wait 500
endif
if insysmsg 'not hold any more'
    overhead 'Stocking shelf...
    lifttype 'keg' 1
    drop 'shelf' -1 -1 0
    overhead 'keg dropped'
    wait 500
endif
#make potion
dclicktype 'mortar and pestle'
waitforgump 949095101
gumpresponse 107
waitforgump 949095101
gumpresponse 202
waitforgump 949095101
waitforgump 949095101
pause 2000
gumpclose
#Uhh... loop.
loop
