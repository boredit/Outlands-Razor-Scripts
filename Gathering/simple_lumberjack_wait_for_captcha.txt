#a simple lumberjack macro
#equip a hatchet
if lhandempty
dclicktype '3908'
wait 1000
endif
#use hatchet for the first time
hotkey 'use item in hand'
wft
target 'self'
wait 500
#begin captcha resilient loop
while not dead
    #world save pause
    if insysmsg 'world is saving'
        wait 4000
    endif
    #turn logs in to boards
    while findtype 7133 true
        usetype 7133 true
        wait 200
    endwhile
    #stop if over weight
    if weight >= 580
        overhead 'You are over weight!' 55
        stop
    endif
    #captcha resilient use of hatchet
    if insysmsg 'chop some'
        hotkey 'use item in hand'
        wft
        target 'self'
    endif
    if insysmsg 'fail'
        hotkey 'use item in hand'
        wft
        target 'self'
    endif
    #stop trying to chop when the node is exhausted
    if insysmsg 'You do not see'
        overhead 'Move along! Move along!' 55
    stop
    endif
wait 500    
endwhile
